**2.1 Script Code (`processRawData`):** ‡πÄ‡∏Ç‡∏µ‡∏¢‡∏ô Logic ‡πÉ‡∏ô GAS ‡πÄ‡∏û‡∏∑‡πà‡∏≠‡∏î‡∏∂‡∏á Raw Data, ‡πÅ‡∏¢‡∏Å Event Type, ‡πÅ‡∏•‡∏∞‡∏ô‡∏±‡∏ö‡∏Ñ‡∏ß‡∏≤‡∏°‡∏ñ‡∏µ‡πà‡πÄ‡∏û‡∏∑‡πà‡∏≠‡∏Ñ‡∏≥‡∏ô‡∏ß‡∏ì Metrics

// ‡∏ü‡∏±‡∏á‡∏Å‡πå‡∏ä‡∏±‡∏ô‡∏ô‡∏µ‡πâ‡∏à‡∏∞‡∏î‡∏∂‡∏á Raw Data ‡∏°‡∏≤‡∏õ‡∏£‡∏∞‡∏°‡∏ß‡∏•‡∏ú‡∏•‡πÅ‡∏•‡∏∞‡∏Ñ‡∏≥‡∏ô‡∏ß‡∏ì Metrics
function processRawData() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const rawSheet = ss.getSheetByName("Raw Data");
  const processedSheet = ss.getSheetByName("Processed Data");
  const rawData = rawSheet.getDataRange().getValues();
  rawData.shift(); // ‡∏•‡∏ö Header

  let metrics = {};
  const today = Utilities.formatDate(new Date(), ss.getSpreadsheetTimeZone(), "yyyy-MM-dd");

  // **Logic Core:** Loop ‡πÄ‡∏û‡∏∑‡πà‡∏≠‡∏ß‡∏¥‡πÄ‡∏Ñ‡∏£‡∏≤‡∏∞‡∏´‡πå Event ‡πÅ‡∏ï‡πà‡∏•‡∏∞‡∏£‡∏≤‡∏¢‡∏Å‡∏≤‡∏£
  rawData.forEach(row => {
    const jsonString = row[1]; // Column B (Raw_JSON)
    try {
      const data = JSON.parse(jsonString);
      if (data.events && data.events.length > 0) {
        data.events.forEach(event => {
          const eventType = event.type;
          const userId = event.source.userId || 'N/A';
          // **Metric 1: Count By Event Type**
          metrics[`${today}_${eventType}`] = (metrics[`${today}_${eventType}`] || 0) + 1;
        });
      }
    } catch (e) {
      Logger.log("Error parsing JSON: " + e);
    }
  });

  // **Save Processed Data**
  for (const key in metrics) {
    const [date, type] = key.split('_');
    processedSheet.appendRow([date, type, 'N/A', metrics[key], 'TOTAL']);
  }
  // **‡πÄ‡∏Ñ‡∏•‡∏µ‡∏¢‡∏£‡πå Raw Data** ‡πÄ‡∏û‡∏∑‡πà‡∏≠‡∏õ‡πâ‡∏≠‡∏á‡∏Å‡∏±‡∏ô‡∏Å‡∏≤‡∏£‡∏õ‡∏£‡∏∞‡∏°‡∏ß‡∏•‡∏ú‡∏•‡∏ã‡πâ‡∏≥ (Optional: rawSheet.clearContents())
}

// ‡πÇ‡∏Ñ‡πâ‡∏î: Webhook Receiver - ‡∏£‡∏±‡∏ö JSON ‡∏î‡∏¥‡∏ö‡∏à‡∏≤‡∏Å Line OA ‡πÅ‡∏•‡∏∞‡∏ö‡∏±‡∏ô‡∏ó‡∏∂‡∏Å‡πÉ‡∏ô‡∏ä‡∏µ‡∏ï Raw Data
function doPost(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const rawSheet = ss.getSheetByName('Raw Data');
  const data = JSON.parse(e.postData.contents);
  rawSheet.appendRow([new Date(), JSON.stringify(data)]);
  return ContentService.createTextOutput(JSON.stringify({'status': 'ok'}));

// ‡πÇ‡∏Ñ‡πâ‡∏î: Data Processor - ‡∏õ‡∏£‡∏∞‡∏°‡∏ß‡∏•‡∏ú‡∏• Raw JSON ‡πÅ‡∏•‡∏∞‡∏ö‡∏±‡∏ô‡∏ó‡∏∂‡∏Å Metrics ‡πÉ‡∏ô‡∏ä‡∏µ‡∏ï Processed Data
function processRawData() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const rawSheet = ss.getSheetByName("Raw Data");
  const processedSheet = ss.getSheetByName("Processed Data");
  const rawData = rawSheet.getDataRange().getValues();
  rawData.shift(); // Remove header
  let metrics = {};
  const today = Utilities.formatDate(new Date(), ss.getSpreadsheetTimeZone(), "yyyy-MM-dd");

  rawData.forEach(row => {
    const jsonString = row[1];
    try {
      const data = JSON.parse(jsonString);
      if (data.events && data.events.length > 0) {
        data.events.forEach(event => {
          const eventType = event.type;
          metrics[`${today}_${eventType}`] = (metrics[`${today}_${eventType}`] || 0) + 1;
        });
      }
    } catch (e) {
      Logger.log("Error parsing JSON: " + e);
    }
  });

  // Save Processed Data
  for (const key in metrics) {
    const [date, type] = key.split('_');
    processedSheet.appendRow([date, type, 'N/A', metrics[key], 'TOTAL']); // (Date, Type, UserID, Count, Dimension)
  }
}
